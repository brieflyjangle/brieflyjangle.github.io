void PreOrderTraverse (BiTree T){//[1[2[3[4[n   ][5[n  ]]] n  ]  6 ]
    if ( T == NULL )             //         n n       n n    n n
        return
    printf("%c",T->data);        //1 2 3 4         5              6
    PreorderTraverse(T->lchild); //2 3 4 n         n              7
    PreorderTraverse(T->rchild); //          n 5    n      n  6  
}

void PostOrderTraverse(BiTree T){
  if(T = NULL)
  return;
  PostOrderTraverse(T->lchild);
  PostOrderTraverse(T->rchild);
  printf("%c",T->data);
}
